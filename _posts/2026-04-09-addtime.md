_This calculator is mainly for furnance time calculation._ 

## Add Time from Now

<label>
  Days:
  <input id="daysNow" type="number" value="0" min="0">
</label>
<br>

<label>
  Hours:
  <input id="hoursNow" type="number" value="0" min="0">
</label>
<br>

<label>
  Minutes:
  <input id="minutesNow" type="number" value="0" min="0">
</label>
<br><br>

<button onclick="addTime()">Calculate</button>

<h3>Result</h3>
<p id="resultNow">—</p>

<script>
function addTime() {
  const days = Number(document.getElementById("daysNow").value || 0);
  const hours = Number(document.getElementById("hoursNow").value || 0);
  const minutes = Number(document.getElementById("minutesNow").value || 0);

  const now = new Date();

  const totalMinutes =
    (days * 24 * 60) +
    (hours * 60) +
    minutes;

  const future = new Date(now.getTime() + totalMinutes * 60000);

  const optionsDate = {
    year: "numeric",
    month: "long",
    day: "numeric",
    weekday: "long"
  };

  const optionsTime = {
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false
  };

  const dateStr = future.toLocaleDateString(undefined, optionsDate);
  const timeStr = future.toLocaleTimeString(undefined, optionsTime);

  document.getElementById("resultNow").textContent =
    `${dateStr}, ${timeStr}`;
}
</script>

## Add Time to Selected Date

<label>
  Start date and time:
  <input id="startTime" type="datetime-local">
</label>

<br><br>

<label>
  Days:
  <input id="daysSelect" type="number" value="0">
</label>
<br>

<label>
  Hours:
  <input id="hoursSelect" type="number" value="0">
</label>
<br>

<label>
  Minutes:
  <input id="minutesSelect" type="number" value="0">
</label>

<br><br>
<button onclick="calculate()">Calculate</button>

<h3>Result</h3>
<p id="resultSelect">—</p>

<script>
function calculate() {
  const startValue = document.getElementById("startTime").value;

  if (!startValue) {
    document.getElementById("resultSelect").textContent =
      "Please select a start date and time.";
    return;
  }

  const days = Number(document.getElementById("daysSelect").value || 0);
  const hours = Number(document.getElementById("hoursSelect").value || 0);
  const minutes = Number(document.getElementById("minutesSelect").value || 0);

  const startDate = new Date(startValue);

  const totalMinutes =
    (days * 24 * 60) +
    (hours * 60) +
    minutes;

  const futureDate = new Date(
    startDate.getTime() + totalMinutes * 60000
  );

  const dateOptions = {
    year: "numeric",
    month: "long",
    day: "numeric",
    weekday: "long"
  };

  const timeOptions = {
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false
  };

  const dateStr = futureDate.toLocaleDateString(undefined, dateOptions);
  const timeStr = futureDate.toLocaleTimeString(undefined, timeOptions);

  document.getElementById("resultSelect").textContent =
    `${dateStr}, ${timeStr}`;
}
</script>
